Function Component Example

// UserCard.jsx
import React, { useState } from 'react';

function UserCard({ user, onEdit, className = '' }) {
    const [showDetails, setShowDetails] = useState(false);
    const [isEditing, setIsEditing] = useState(false);

    const handleEdit = () => {
        setIsEditing(true);
        onEdit?.(user);
    };

    return (
        <div className={`bg-white rounded-lg shadow-md p-6 ${className}`}>
            <div className="flex items-center justify-between mb-4">
                <h3 className="text-xl font-semibold text-gray-900">
                    {user.name}
                </h3>
                <button
                    onClick={handleEdit}
                    className="text-blue-600 hover:text-blue-800"
                >
                    Edit
                </button>
            </div>
            
            <p className="text-gray-600 mb-4">{user.email}</p>
            
            <button
                onClick={() => setShowDetails(!showDetails)}
                className="bg-blue-500 text-white px-4 py-2 rounded hover:bg-blue-600"
            >
                {showDetails ? 'Hide' : 'Show'} Details
            </button>
            
            {showDetails && (
                <div className="mt-4 p-4 bg-gray-50 rounded">
                    <p><strong>Role:</strong> {user.role}</p>
                    <p><strong>Joined:</strong> {user.joinedDate}</p>
                    <p><strong>Status:</strong> 
                        <span className={`px-2 py-1 rounded text-sm ${
                            user.isActive ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'
                        }`}>
                            {user.isActive ? 'Active' : 'Inactive'}
                        </span>
                    </p>
                </div>
            )}
        </div>
    );
}

export default UserCard;

Custom Hooks for Reusability

// hooks/useUserCard.js
import { useState, useCallback } from 'react';

export function useUserCard(initialUser) {
    const [user, setUser] = useState(initialUser);
    const [showDetails, setShowDetails] = useState(false);
    const [isLoading, setIsLoading] = useState(false);

    const toggleDetails = useCallback(() => {
        setShowDetails(prev => !prev);
    }, []);

    const updateUser = useCallback(async (updates) => {
        setIsLoading(true);
        try {
            // API call simulation
            await new Promise(resolve => setTimeout(resolve, 1000));
            setUser(prev => ({ ...prev, ...updates }));
        } catch (error) {
            console.error('Failed to update user:', error);
        } finally {
            setIsLoading(false);
        }
    }, []);

    return {
        user,
        showDetails,
        isLoading,
        toggleDetails,
        updateUser
    };
}

// Usage in component
function UserCard({ initialUser }) {
    const { user, showDetails, isLoading, toggleDetails, updateUser } = useUserCard(initialUser);
    
    // Component JSX...
}

Vue.js Components

Progressive framework with single-file components

  • • Easy learning curve

<!-- UserCard.vue (Options API) -->
<template>
  <div :class="['user-card', className]">
    <h3>{{ user.name }}</h3>
    <p>{{ user.email }}</p>
    
    <button @click="toggleDetails">
      {{ showDetails ? 'Hide' : 'Show' }} Details
    </button>
    
    <div v-show="showDetails" class="details">
      <p>Role: {{ user.role }}</p>
      <p>Status: {{ userStatus }}</p>
    </div>
  </div>
</template>

<script>
export default {
  name: 'UserCard',
  props: {
    user: {
      type: Object,
      required: true,
      validator(value) {
        return value && value.name && value.email
      }
    },
    className: {
      type: String,
      default: ''
    }
  },
  emits: ['edit', 'delete'],
  data() {
    return {
      showDetails: false,
      isLoading: false
    }
  },
  computed: {
    userStatus() {
      return this.user.isActive ? 'Active' : 'Inactive'
    },
    cardClasses() {
      return [
        'bg-white rounded-lg shadow-md p-6',
        this.className,
        { 'opacity-50': this.isLoading }
      ]
    }
  },
  methods: {
    toggleDetails() {
      this.showDetails = !this.showDetails
    },
    handleEdit() {
      this.$emit('edit', this.user)
    },
    async updateStatus() {
      this.isLoading = true
      try {
        // Simulate API call
        await new Promise(resolve => setTimeout(resolve, 1000))
        this.user.isActive = !this.user.isActive
      } finally {
        this.isLoading = false
      }
    }
  },
  watch: {
    'user.name': {
      handler(newName) {
        console.log(`User name changed to: ${newName}`)
      },
      immediate: true
    }
  }
}
</script>

Alpine.js Components

Lightweight reactive framework for HTML

✅ Strengths

  • • No build process required
  • • Minimal learning curve
  • • Works with existing HTML
  • • Very lightweight (~10kb)
  • • Great for progressive enhancement

⚠️ Considerations

  • • Limited component abstraction
  • • No built-in router
  • • Template reuse can be verbose
  • • Less suitable for complex SPAs

Alpine.data() Components

// Define reusable component logic
Alpine.data('userCard', (user) => ({
    user: user,
    showDetails: false,
    isLoading: false,
    
    init() {
        console.log('User card initialized for:', this.user.name);
    },
    
    toggleDetails() {
        this.showDetails = !this.showDetails;
    },
    
    async updateStatus() {
        this.isLoading = true;
        
        try {
            // Simulate API call
            await new Promise(resolve => setTimeout(resolve, 1000));
            this.user.isActive = !this.user.isActive;
        } catch (error) {
            console.error('Failed to update status:', error);
        } finally {
            this.isLoading = false;
        }
    },
    
    handleEdit() {
        this.$dispatch('user-edit', { user: this.user });
    },
    
    get userStatusClass() {
        return this.user.isActive 
            ? 'bg-green-100 text-green-800' 
            : 'bg-red-100 text-red-800';
    },
    
    get formattedJoinDate() {
        return new Date(this.user.joinedDate).toLocaleDateString();
    }
}));

HTMX Components

Server-side rendered components with AJAX interactions

✅ Strengths

  • • Server-side rendering (SEO friendly)

Laravel Backend Component

<?php
// app/Http/Controllers/UserCardController.php
namespace App\Http\Controllers;

use App\Models\User;
use Illuminate\Http\Request;

class UserCardController extends Controller
{
    public function show(User $user)
    {
        return view('components.user-card', compact('user'));
    }
    
    public function details(User $user)
    {
        return view('components.user-card-details', compact('user'));
    }
    
    public function toggleStatus(User $user)
    {
        $user->update(['is_active' => !$user->is_active]);
        
        return view('components.user-status', compact('user'));
    }
    
    public function edit(User $user)
    {
        return view('components.user-edit-form', compact('user'));
    }
    
    public function update(Request $request, User $user)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:255',
            'email' => 'required|email|unique:users,email,' . $user->id,
            'role' => 'required|string'
        ]);
        
        $user->update($validated);
        
        return view('components.user-card', compact('user'));
    }
}

    public string $name;
    public string $email;
    public string $role;
    
    protected $rules = [
        'name' => 'required|string|max:255',
        'email' => 'required|email',
        'role' => 'required|string'
    ];
    
    public function mount(User $user)
    {
        $this->user = $user;
        $this->name = $user->name;
        $this->email = $user->email;
        $this->role = $user->role;
    }
    
    public function toggleDetails()
    {
        $this->showDetails = !$this->showDetails;
        
        // Emit event for analytics or other components
        $this->emit('userDetailsToggled', $this->user->id, $this->showDetails);
    }
    
    public function startEdit()
    {
        $this->isEditing = true;
        $this->resetErrorBag();
    }
    
    public function cancelEdit()
    {
        $this->isEditing = false;
        
        // Reset to original values
        $this->name = $this->user->name;
        $this->email = $this->user->email;
        $this->role = $this->user->role;
        
        $this->resetErrorBag();
    }
    
    public function saveUser()
    {
        $this->validate();
        
        $this->isLoading = true;
        
        try {
            $this->user->update([
                'name' => $this->name,
                'email' => $this->email,
                'role' => $this->role
            ]);
            
            $this->isEditing = false;
            
            // Show success message
            session()->flash('message', 'User updated successfully!');
            
            // Emit event to parent component or other listeners
            $this->emit('userUpdated', $this->user->id);
            
        } catch (\Exception $e) {
            $this->addError('general', 'Failed to update user: ' . $e->getMessage());
        } finally {
            $this->isLoading = false;
        }
    }
    
    public function toggleStatus()
    {
        $this->isLoading = true;
        
        try {
            $this->user->update([
                'is_active' => !$this->user->is_active
            ]);
            
            // Refresh the user model
            $this->user->refresh();
            
        } catch (\Exception $e) {
            $this->addError('status', 'Failed to update status');
        } finally {
            $this->isLoading = false;
        }
    }
    
    public function deleteUser()
    {
        // Emit event to parent for confirmation
        $this->emit('confirmUserDeletion', $this->user->id);
    }
    
    // Computed property
    public function getUserStatusClassProperty()
    {
        return $this->user->is_active 
            ? 'bg-green-100 text-green-800' 
            : 'bg-red-100 text-red-800';
    }
    
    // Lifecycle hooks
    public function updatedName($value)
    {
        $this->name = ucwords($value);
    }
    
    public function updatedEmail($value)
    {
        $this->email = strtolower($value);
    }
    
    public function render()
    {
        return view('livewire.user-card');
    }
}

Usage in Parent Component

<?php
// app/Http/Livewire/UserDirectory.php
namespace App\Http\Livewire;

use App\Models\User;
use Livewire\Component;
use Livewire\WithPagination;

class UserDirectory extends Component
{
    use WithPagination;
    
    public string $search = '';
    public string $sortBy = 'name';
    public string $sortDirection = 'asc';
    
    protected $listeners = [
        'userUpdated' => 'handleUserUpdated',
        'confirmUserDeletion' => 'confirmDeletion'
    ];
    
    public function updatingSearch()
    {
        $this->resetPage();
    }
    
    public function sortBy($field)
    {
        if ($this->sortBy === $field) {
            $this->sortDirection = $this->sortDirection === 'asc' ? 'desc' : 'asc';
        } else {
            $this->sortBy = $field;
            $this->sortDirection = 'asc';
        }
    }
    
    public function handleUserUpdated($userId)
    {
        session()->flash('message', 'User updated successfully!');
        // Optionally refresh user data or emit other events
    }
    
    public function confirmDeletion($userId)
    {
        // You could emit a browser event to show a confirmation modal
        $this->emit('showDeleteConfirmation', $userId);
    }
    
    public function render()
    {
        $users = User::query()
            ->when($this->search, function ($query) {
                $query->where('name', 'like', '%' . $this->search . '%')
                      ->orWhere('email', 'like', '%' . $this->search . '%');
            })
            ->orderBy($this->sortBy, $this->sortDirection)
            ->paginate(12);
            
        return view('livewire.user-directory', compact('users'));
    }
}
